const xhr = new XMLHttpRequest(); xhr.open("GET", "https://jsonplaceholder.typicode.com/users", true); xhr.onreadystatechange = function () { if (xhr.readyState === 4 && xhr.status === 200) { const users = JSON.parse(xhr.responseText); const userNames = users.map((user) => user.name); console.log(userNames); document.body.innerHTML = `
    ${userNames .map((name) => `
  • ${name}
    • `) .join("")}
`; } }; xhr.send();
